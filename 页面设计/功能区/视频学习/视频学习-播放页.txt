    function toggleDescription() {
        const desc = document.getElementById('video-description');
        const text = document.getElementById('toggle-text');
        const icon = document.getElementById('toggle-icon');
        
        if (desc.classList.contains('line-clamp-2')) {
            desc.classList.remove('line-clamp-2');
            text.innerText = '收起描述';
            icon.innerText = 'expand_less';
        } else {
            desc.classList.add('line-clamp-2');
            text.innerText = '展开更多';
            icon.innerText = 'expand_more';
        }
    }

    // Touch feedback for buttons
    document.querySelectorAll('button').forEach(btn => {
        btn.addEventListener('touchstart', () => btn.classList.add('opacity-70'));
        btn.addEventListener('touchend', () => btn.classList.remove('opacity-70'));
    });
</script>
