        function switchTab(btn) {
            const tabs = btn.parentNode.querySelectorAll('button');
            tabs.forEach(tab => {
                tab.classList.remove('bg-primary', 'text-on-primary', 'shadow-sm');
                tab.classList.add('bg-surface-container', 'text-on-surface-variant');
            });
            btn.classList.remove('bg-surface-container', 'text-on-surface-variant');
            btn.classList.add('bg-primary', 'text-on-primary', 'shadow-sm');
            
            // Interaction: Simulate list refresh
            const list = document.getElementById('bank-list');
            list.style.opacity = '0.5';
            setTimeout(() => {
                list.style.opacity = '1';
            }, 300);
        }

        // Simple haptic-like feedback for buttons
        document.querySelectorAll('button, a').forEach(el => {
            el.addEventListener('click', () => {
                if (window.navigator && window.navigator.vibrate) {
                    window.navigator.vibrate(5);
                }
            });
        });
    </script>
