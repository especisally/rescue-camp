<script>
        let isLoggedIn = false;

        function toggleLoginState() {
            isLoggedIn = !isLoggedIn;
            const guestAvatar = document.getElementById('guest-avatar');
            const userAvatar = document.getElementById('user-avatar');
            const guestText = document.getElementById('profile-text-guest');
            const userText = document.getElementById('profile-text-user');
            const logoutBtn = document.getElementById('logout-btn');

            if (isLoggedIn) {
                guestAvatar.classList.add('hidden');
                userAvatar.classList.remove('hidden');
                guestText.classList.add('hidden');
                userText.classList.remove('hidden');
                logoutBtn.classList.remove('hidden');
            } else {
                guestAvatar.classList.remove('hidden');
                userAvatar.classList.add('hidden');
                guestText.classList.remove('hidden');
                userText.classList.add('hidden');
                logoutBtn.classList.add('hidden');
            }
        }
    </script>
